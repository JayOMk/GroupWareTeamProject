CREATE SEQUENCE msg_seq
  START WITH 1
  INCREMENT BY 1
  NOMAXVALUE;
  
  CREATE TABLE msg (
  msgid NUMBER PRIMARY KEY,
  to_id VARCHAR2(255),
  from_id VARCHAR2(255),
  title VARCHAR2(255),
  content VARCHAR2(4000),
  time DATE,
  hit NUMBER
);

CREATE OR REPLACE TRIGGER msg_trigger
BEFORE INSERT ON msg
FOR EACH ROW
BEGIN
  SELECT msg_seq.NEXTVAL INTO :NEW.msgid FROM DUAL;
END;
